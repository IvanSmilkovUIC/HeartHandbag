#include <IRremote.h>

IRsend irsend;
int PulseSensor = A0;
int LED13 = 13;
int Signal;
int Threshold = 530;

void setup() {
   pinMode(3, OUTPUT);
   Serial.begin(115200);

}

void loop() {

   Signal = analogRead(PulseSensor);                                    

   Serial.println(Signal);

   if(Signal > Threshold){ 
     irsend.sendNEC(0xFFA25D, 32);
     delay(150);
   }

delay(75);

}
